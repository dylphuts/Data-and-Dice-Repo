[---SUBROUTINE: PRINT STRING ---]
[ Entry point: 95]
T 95 K        [ start routine at slot 95]
G K           [ jump to start of logic]
T 47 K        [ save workspace]
P 20 @        [ printer control]
T Z           [ save printer control]

A 3 F         [ load return ticket]
T M           [ save return address]

[ initialize string pointer]
A 1 F         [ base address of string]
A 2 M         [ add current offset]
T 5 @         [ save pointer]

[ loop start at absolute slot, e.g., 100]
100 A 5 @     [ load next character]
A F           [ fetch character]
U 4 M         [ save for printing]
S 32 F        [ check for terminator]
G 120         [ jump to return if terminator]

O 4 M         [ print character]
T F           [ clear accumulator]

A 5 @         [ increment pointer]
A 1 M         [ next memory word]
T 5 @         [ save pointer]
G 100         [ loop back]

120 E M       [ restore return ticket and return]
