< Exp >    ::= < Exp > + < Term >
               | < Exp > - < Term >
               | < Term >

< Term >   ::= < Term > * < Factor >
               | < Term > / < Factor >
               | < Factor>

< Factor > ::= < Base > ^ < Factor >
               | < Base >

< Base >   ::= - <Base>
               |( < Exp > )
               | < Number >

< Number > ::= [0-9]+
               | [0-9]+ . [0-9]+  

< variable > ::= <Base>
